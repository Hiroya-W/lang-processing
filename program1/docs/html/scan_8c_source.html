62  } else if (current_char == '\r' || current_char == '\n') { /* End of Line */
63  if (current_char == '\r') {

66  }

70  if (next_char == '\r') {

72  }

75  }
76  } else if (_isblank(current_char)) { /* Separator (Space or Tab) */
77  look_ahead();

79  return -1;
80  } else if (isalpha(current_char)) { /* Name or Keyword */
